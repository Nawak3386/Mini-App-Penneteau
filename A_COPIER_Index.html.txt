<script>
const STORAGE_KEY = "feuille_heures_penneteau_v4";
const SALARIE_KEY = "dernierSalarie";
const PRESENT_STATUS = "Pr\u00e9sent";
const STATUSES = [PRESENT_STATUS, "Absent", "Cong\u00e9s", "Maladie", "Formation", "Intemp\u00e9ries"];

const jours = [
  {jour:"Lundi", debut:"08:00", fin:"16:30", pause:"30", vendredi:false, prevu:480},
  {jour:"Mardi", debut:"08:00", fin:"16:30", pause:"30", vendredi:false, prevu:480},
  {jour:"Mercredi", debut:"08:00", fin:"16:30", pause:"30", vendredi:false, prevu:480},
  {jour:"Jeudi", debut:"08:00", fin:"16:30", pause:"30", vendredi:false, prevu:480},
  {jour:"Vendredi", debut:"08:00", fin:"12:00", pause:"0", vendredi:true, prevu:240}
];

function min(t){
  if(!t) return 0;
  let p=t.split(":");
  return (+p[0]||0)*60+(+p[1]||0);
}

function formatMinutes(minutes){
  minutes = Math.round(Math.abs(minutes));
  const h = Math.floor(minutes / 60);
  const m = minutes % 60;
  return h + "h" + String(m).padStart(2, "0");
}

function calculZone(km){
  if(km === "" || km === null || km === undefined){
    return "-";
  }

  km = Number(km);

  if(km <= 5) return "Zone 1A";
  if(km <= 10) return "Zone 1B";
  if(km <= 20) return "Zone 2";
  if(km <= 30) return "Zone 3";
  if(km <= 40) return "Zone 4";
  if(km <= 50) return "Zone 5";
  return "Hors zone";
}

function pauseLabel(value){
  if(value === "0") return "Aucune pause";
  if(value === "30") return "30 min";
  if(value === "60") return "1 h";
  return value + " min";
}

function appliquerEcart(compteurs, ecart){
  ecart = Number(ecart) || 0;

  if(ecart < 0){
    let dette = Math.abs(ecart);

    if(compteurs.heuresSup > 0){
      const compensation = Math.min(dette, compteurs.heuresSup);
      compteurs.heuresSup -= compensation;
      dette -= compensation;
    }

    compteurs.aDevoir += dette;
    return;
  }

  if(ecart <= 0) return;

  if(compteurs.aDevoir > 0){
    const deduction = Math.min(ecart, compteurs.aDevoir);
    compteurs.aDevoir -= deduction;
    ecart -= deduction;
  }

  compteurs.heuresSup += ecart;
}

function getResumeSemaine(){
  return {
    totalTravaille: document.getElementById("totalSemaine").textContent,
    aDevoir: document.getElementById("aDevoir").textContent,
    aDeduire: document.getElementById("aDeduire").textContent,
    heuresSupplementaires: document.getElementById("solde").textContent
  };
}

  const box=document.getElementById("days");
  box.innerHTML = "";

  jours.forEach(j=>{
    box.innerHTML += `
    <div class="card day" data-jour="${j.jour}" data-prevu="${j.prevu}">
      <h3>${j.jour}</h3>

      <label>Statut
        <select class="statut">
          ${STATUSES.map(statut => `<option value="${statut}">${statut}</option>`).join("")}
        </select>
      </label>

      <label>Date du jour
        <input type="date" class="date">
      </label>

      <div class="presence-fields">
      <label>Nom du chantier
        <input class="chantier" placeholder="Nom du chantier">
      </label>

      <div class="two">
        <label>D\u00e9but r\u00e9el
          <input type="time" class="debut" value="${j.debut}">
        </label>

        <label>Fin r\u00e9elle
          <input type="time" class="fin" value="${j.fin}">
        </label>
      </div>

      <label>Pause d\u00e9jeuner
        <select class="pause">
          <option value="30" ${j.pause==="30" ? "selected" : ""}>30 min</option>
          <option value="60" ${j.pause==="60" ? "selected" : ""}>1 h</option>
          <option value="0" ${j.pause==="0" ? "selected" : ""}>Aucune pause</option>
        </select>
      </label>

      ${j.vendredi ? `
      <div class="friday">
        Vendredi pr\u00e9vu : 08h00 \u2192 12h00, sans pause = 4h00.
        <label class="checkline">
          <input type="checkbox" class="exception" onchange="toggleFriday(this)">
          Apr\u00e8s-midi exceptionnel
        </label>
      </div>` : ""}

      <label>Distance entreprise \u2192 chantier (km)
        <input type="number" class="distanceKm" min="0" step="0.1" placeholder="Ex : 18">
      </label>

      <label>Zone d\u00e9placement
        <input class="zone" readonly>
      </label>
      </div>

      <label>Note
        <input class="note" placeholder="Optionnel">
      </label>

      <p><b>Total r\u00e9el : <span class="totalJour">0h00</span></b></p>
      <p><b>Pr\u00e9vu : <span class="prevuJour">${formatMinutes(j.prevu)}</span></b></p>
      <p class="ecart"><span class="ecartJour ok">Non rempli</span></p>
    </div>`;
  });

  document.querySelectorAll("input,select").forEach(e=>{
    e.addEventListener("input",()=>{calculer(); sauvegarderAuto();});
    e.addEventListener("change",()=>{calculer(); sauvegarderAuto();});
  });

  document.getElementById("salarie").addEventListener("change", function(){
    localStorage.setItem(SALARIE_KEY, this.value);
    sauvegarderAuto();
  });

  restaurer();
  document.querySelectorAll(".day").forEach(updateDayVisibility);
  calculer();
}

function getAbsenceStatus(c){
  return c.querySelector(".statut").value || PRESENT_STATUS;
}

function isPresent(c){
  return getAbsenceStatus(c) === PRESENT_STATUS;
}

function updateDayVisibility(c){
  const present = isPresent(c);
  c.querySelector(".presence-fields").classList.toggle("hidden", !present);
  c.querySelector(".prevuJour").closest("p").classList.toggle("hidden", !present);

  // Une absence masque les champs de travail et neutralise tous les calculs.
  if(!present){
    c.querySelector(".zone").value = "";
    c.querySelector(".totalJour").textContent = "0h00";
    c.querySelector(".ecartJour").textContent = getAbsenceStatus(c);
    c.querySelector(".ecartJour").className = "ecartJour ok";
  }
}

function toggleFriday(cb){
  const c=cb.closest(".day");
  const fin=c.querySelector(".fin");
  const pause=c.querySelector(".pause");

  if(cb.checked){
    if(fin.value==="12:00") fin.value="16:30";
    pause.value="30";
  } else {
    fin.value="12:00";
    pause.value="0";
  }

  calculer();
  sauvegarderAuto();
}

function jourRempli(c){
  return !!(
    c.querySelector(".date").value ||
    c.querySelector(".chantier").value.trim() ||
    !isPresent(c)
  );
}

function totalCardMinutes(c){
  if(!jourRempli(c) || !isPresent(c)){
    return 0;
  }

  const debut = c.querySelector(".debut").value;
  const fin = c.querySelector(".fin").value;
  const pause = Number(c.querySelector(".pause").value) || 0;

  let minutes = min(fin) - min(debut) - pause;
  if(minutes < 0) minutes = 0;

  return minutes;
}

function calculer(){
  let total=0;
  const compteurs = {aDevoir:0, heuresSup:0};
  let heuresEnPlusBrutes=0;

  document.querySelectorAll(".day").forEach(c=>{
    updateDayVisibility(c);

    const h = totalCardMinutes(c);
    const prevu = Number(c.dataset.prevu);
    const ecart = h - prevu;

    c.querySelector(".totalJour").textContent = formatMinutes(h);
    if(isPresent(c)){
      c.querySelector(".zone").value = calculZone(c.querySelector(".distanceKm").value);
    }

    const ecartBox = c.querySelector(".ecartJour");

    if(!jourRempli(c)){
      ecartBox.textContent = "Non rempli";
      ecartBox.className = "ecartJour ok";
      return;
    }

    if(!isPresent(c)){
      ecartBox.textContent = getAbsenceStatus(c);
      ecartBox.className = "ecartJour ok";
      return;
    }

    total += h;

    if(ecart < 0){
      appliquerEcart(compteurs, ecart);
      ecartBox.textContent = "\u2796 " + formatMinutes(ecart) + " \u00e0 devoir";
      ecartBox.className = "ecartJour moins";
    } else if(ecart > 0){
      heuresEnPlusBrutes += ecart;
      appliquerEcart(compteurs, ecart);
      ecartBox.textContent = "\u2795 " + formatMinutes(ecart) + " \u00e0 d\u00e9duire";
      ecartBox.className = "ecartJour plus";
    } else {
      ecartBox.textContent = "OK";
      ecartBox.className = "ecartJour ok";
    }
  });

  totalSemaine.textContent = formatMinutes(total);
  document.getElementById("aDevoir").textContent = "\u2796 " + formatMinutes(compteurs.aDevoir);
  document.getElementById("aDeduire").textContent = "\u2795 " + formatMinutes(heuresEnPlusBrutes);
  solde.textContent = formatMinutes(compteurs.heuresSup);
}

function buildFeuilleId(data){
  // M\u00eame salari\u00e9 + m\u00eame p\u00e9riode + m\u00eames dates = m\u00eame feuille, donc remplacement fiable.
  const dates = data.lignes
    .map(ligne => ligne.date)
    .filter(Boolean)
    .sort()
    .join("_");

  return [data.salarie, data.semaine.trim(), dates].join("|");
}

function buildPeriodeDepuisDates(lignes){
  const dates = lignes
    .map(ligne => ligne.date)
    .filter(Boolean)
    .sort();

  if(dates.length === 0) return "";
  if(dates.length === 1) return dates[0];
  return dates[0] + " au " + dates[dates.length - 1];
}

function collecterData(){
  const data={
    salarie: document.getElementById("salarie").value,
    semaine: "",
    baseContrat: "36",
    resume:getResumeSemaine(),
    lignes: []
  };

  document.querySelectorAll(".day").forEach(c=>{
    const date = c.querySelector(".date").value;
    const chantier = c.querySelector(".chantier").value.trim();
    const statut = getAbsenceStatus(c);
    const present = statut === PRESENT_STATUS;

    if(!jourRempli(c)){
      return;
    }

    const totalMinutes = present ? totalCardMinutes(c) : 0;
    const prevu = Number(c.dataset.prevu);
    const ecart = present ? totalMinutes - prevu : 0;

    let typeHeure = "OK";
    let heuresSup = "0h00";
    let ecartAffiche = "0h00";

    if(ecart < 0){
      typeHeure = "\u2796 \u00c0 devoir";
      ecartAffiche = "\u2796 " + formatMinutes(ecart);
    } else if(ecart > 0){
      typeHeure = "\u2795 \u00c0 d\u00e9duire";
      heuresSup = formatMinutes(ecart);
      ecartAffiche = "\u2795 " + formatMinutes(ecart);
    }

    data.lignes.push({
      jour:c.dataset.jour,
      date:date,
      statut:statut,
      debut:present ? c.querySelector(".debut").value : "",
      fin:present ? c.querySelector(".fin").value : "",
      pause:present ? pauseLabel(c.querySelector(".pause").value) : "",
      pauseMinutes:present ? c.querySelector(".pause").value : "0",
      typeHeure:typeHeure,
      heuresSup:heuresSup,
      totalJour:present ? formatMinutes(totalMinutes) : "",
      chantier:present ? chantier : "",
      distanceKm:present ? c.querySelector(".distanceKm").value : "",
      zone:present ? c.querySelector(".zone").value : "",
      note:c.querySelector(".note").value,
      ecart:ecartAffiche
    });
  });

  data.semaine = buildPeriodeDepuisDates(data.lignes);
  data.feuilleId = buildFeuilleId(data);
  return data;
}

function collecterFormulaireComplet(){
  const data={
    salarie: document.getElementById("salarie").value,
    semaine: "",
    baseContrat: "36",
    resume:getResumeSemaine(),
    jours:[]
  };

  document.querySelectorAll(".day").forEach(c=>{
    data.jours.push({
      jour:c.dataset.jour,
      statut:getAbsenceStatus(c),
      date:c.querySelector(".date").value,
      chantier:c.querySelector(".chantier").value,
      debut:c.querySelector(".debut").value,
      fin:c.querySelector(".fin").value,
      pause:c.querySelector(".pause").value,
      exception:c.querySelector(".exception") ? c.querySelector(".exception").checked : false,
      distanceKm:c.querySelector(".distanceKm").value,
      note:c.querySelector(".note").value
    });
  });

  return data;
}

function sauvegarderAuto(){
  const data = collecterFormulaireComplet();
  localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
  localStorage.setItem(SALARIE_KEY, document.getElementById("salarie").value);
}

function sauvegarder(){
  sauvegarderAuto();
  status.textContent = "Sauvegard\u00e9 sur cet appareil";
}

function restaurer(){
  const dernierSalarie = localStorage.getItem(SALARIE_KEY);
  if(dernierSalarie){
    document.getElementById("salarie").value = dernierSalarie;
  }

  const saved = localStorage.getItem(STORAGE_KEY);
  if(!saved) return;

  try{
    const data = JSON.parse(saved);

    document.getElementById("salarie").value = data.salarie || dernierSalarie || "J\u00e9r\u00e9my";
    document.querySelectorAll(".day").forEach((c,i)=>{
      const j = data.jours && data.jours[i];
      if(!j) return;

      c.querySelector(".statut").value = j.statut || PRESENT_STATUS;
      c.querySelector(".date").value = j.date || "";
      c.querySelector(".chantier").value = j.chantier || "";
      c.querySelector(".debut").value = j.debut || jours[i].debut;
      c.querySelector(".fin").value = j.fin || jours[i].fin;
      c.querySelector(".pause").value = j.pause || jours[i].pause;
      c.querySelector(".distanceKm").value = j.distanceKm || "";
      c.querySelector(".note").value = j.note || "";

      if(c.querySelector(".exception")){
        c.querySelector(".exception").checked = !!j.exception;
      }

      updateDayVisibility(c);
    });

    status.textContent = "\u2705 Semaine restaur\u00e9e depuis cet appareil";
  }catch(e){
    status.textContent = "\u26a0\ufe0f Sauvegarde locale illisible";
  }
}

function nouvelleSemaine(){
  if(!confirm("Vider la semaine en cours sur cet appareil ?")) return;

  localStorage.removeItem(STORAGE_KEY);

  document.querySelectorAll(".day").forEach((c, i)=>{
    const j = jours[i];

    c.querySelector(".statut").value = PRESENT_STATUS;
    c.querySelector(".date").value = "";
    c.querySelector(".chantier").value = "";
    c.querySelector(".debut").value = j.debut;
    c.querySelector(".fin").value = j.fin;
    c.querySelector(".pause").value = j.pause;
    c.querySelector(".distanceKm").value = "";
    c.querySelector(".zone").value = "";
    c.querySelector(".note").value = "";

    if(c.querySelector(".exception")){
      c.querySelector(".exception").checked = false;
    }

    updateDayVisibility(c);
  });

  calculer();
  status.textContent = "Nouvelle semaine pr\u00eate";
}

function validerAvantEnvoi(data){
  // Validation c\u00f4t\u00e9 navigateur pour donner un message imm\u00e9diat avant l'appel Apps Script.
  if(!data.salarie){
    return "S\u00e9lectionne un salari\u00e9.";
  }

  if(data.lignes.length === 0){
    return "Aucune journ\u00e9e remplie \u00e0 envoyer.";
  }

  for(const ligne of data.lignes){
    const jour = ligne.jour || "une journ\u00e9e";

    if(!ligne.date){
      return "Renseigne la date pour " + jour + ".";
    }

    if(ligne.statut !== PRESENT_STATUS){
      continue;
    }

    if(!ligne.chantier){
      return "Renseigne le chantier pour " + jour + ".";
    }

    if(!ligne.debut || !ligne.fin){
      return "Renseigne les horaires pour " + jour + ".";
    }

    const debut = min(ligne.debut);
    const fin = min(ligne.fin);
    const pause = Number(ligne.pauseMinutes) || 0;

    if(fin <= debut){
      return "L'heure de fin doit \u00eatre sup\u00e9rieure \u00e0 l'heure de d\u00e9but pour " + jour + ".";
    }

    if(fin - debut - pause < 0){
      return "La pause d\u00e9passe le temps travaill\u00e9 pour " + jour + ".";
    }
  }

  return "";
}

function envoyer(){
  calculer();

  const data = collecterData();

  const erreur = validerAvantEnvoi(data);
  if(erreur){
    status.textContent="\u26a0\ufe0f " + erreur;
    alert(erreur);
    return;
  }

  const sendButton = document.getElementById("sendButton");
  sendButton.disabled = true;
  status.textContent="\u23f3 Envoi en cours...";

  google.script.run
    .withSuccessHandler((soldeFinal)=>{
  const message = soldeFinal
    ? "\u2705 Envoy\u00e9 au Google Sheets \u2014 Solde cumul\u00e9 : " + soldeFinal
    : "\u2705 Envoy\u00e9 au Google Sheets";

  document.getElementById("status").textContent = message;
  alert(message);
  sauvegarderAuto();
  sendButton.disabled = false;
})
    .withFailureHandler(err=>{
      status.textContent="\u274c Erreur : " + err.message;
      sendButton.disabled = false;
    })
    .enregistrerSemaine(data);
}

function recalculerSoldes(){
  if(!confirm("Recalculer tous les soldes depuis les feuilles J\u00e9r\u00e9my et Damien ?")) return;

  const recalcButton = document.getElementById("recalcButton");
  recalcButton.disabled = true;
  status.textContent = "Recalcul des soldes en cours...";

  google.script.run
    .withSuccessHandler(message=>{
      const texte = "Recalcul termin\u00e9.\n" + message;
      status.textContent = "Recalcul termin\u00e9.";
      alert(texte);
      recalcButton.disabled = false;
    })
    .withFailureHandler(err=>{
      status.textContent = "Erreur recalcul : " + err.message;
      recalcButton.disabled = false;
    })
    .recalculerTousLesSoldes();
}

setInterval(() => {
  sauvegarderAuto();
}, 5000);

window.addEventListener("beforeunload", () => {
  sauvegarderAuto();
});

build();
