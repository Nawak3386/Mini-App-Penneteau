function normaliserTexte(valeur) {
  return String(valeur || "").trim();
}

function getNomSalarie(valeur) {
  const salarie = normaliserTexte(valeur);
  return APP_CONFIG.employeeSheets[salarie] ? salarie : "";
}

function getNomOngletSalarie(salarie) {
  return APP_CONFIG.employeeSheets[salarie];
}

function convertirHeureEnMinutes(valeur) {
  if (!valeur) return 0;

  let texte = String(valeur)
    .trim()
    .replace("\u2796", "-")
    .replace("\u2795", "")
    .replace("\u00e0 devoir", "")
    .replace("\u00e0 d\u00e9duire", "")
    .replace("sup", "")
    .replace(/\s/g, "");

  const negatif = texte.startsWith("-");
  texte = texte.replace("-", "");

  const parts = texte.split("h");
  const h = Number(parts[0]) || 0;
  const m = Number(parts[1]) || 0;

  const total = h * 60 + m;
  return negatif ? -total : total;
}

function convertirHoraireEnMinutes(valeur) {
  const texte = normaliserTexte(valeur);
  if (!/^\d{2}:\d{2}$/.test(texte)) return null;

  const parts = texte.split(":");
  const heures = Number(parts[0]);
  const minutes = Number(parts[1]);

  if (heures < 0 || heures > 23 || minutes < 0 || minutes > 59) return null;
  return heures * 60 + minutes;
}

function formatHeures(minutes) {
  minutes = Math.max(0, Math.round(Number(minutes) || 0));
  const h = Math.floor(minutes / 60);
  const m = minutes % 60;
  return h + "h" + String(m).padStart(2, "0");
}

function formatSoldeADevoir(minutes) {
  return minutes === 0 ? "0h00" : "-" + formatHeures(minutes);
}

function getOrCreateSheet(ss, sheetName, headers) {
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  if (sheet.getLastRow() === 0 && headers && headers.length) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.setFrozenRows(1);
  }

  return sheet;
}

function getEmployeeSheet(ss, salarie) {
  return getOrCreateSheet(ss, getNomOngletSalarie(salarie), APP_CONFIG.headers);
}

function trouverLigneSolde(sheet, salarie) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return 0;

  const values = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
  for (let i = 0; i < values.length; i++) {
    if (values[i][0] === salarie) {
      return i + 2;
    }
  }

  return 0;
}

function majSoldes(salarie, aDevoir, heuresSup) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = getSoldesSheet(ss);
  let row = trouverLigneSolde(sheet, salarie);

  if (!row) {
    row = sheet.getLastRow() + 1;
    sheet.getRange(row, 1).setValue(salarie);
  }

  sheet.getRange(row, 2, 1, 2).setValues([[
    formatSoldeADevoir(aDevoir),
    formatHeures(heuresSup)
  ]]);
}

function assurerColonnesDepartSoldes(sheet) {
  const headers = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), 5)).getValues()[0];
  if (!headers[3]) sheet.getRange(1, 4).setValue("D\u00e9part \u00e0 devoir");
  if (!headers[4]) sheet.getRange(1, 5).setValue("D\u00e9part heures sup");
}

function getCompteursDepartSoldes(ss, salarie) {
  const sheet = getSoldesSheet(ss);
  assurerColonnesDepartSoldes(sheet);

  let row = trouverLigneSolde(sheet, salarie);
  if (!row) {
    row = sheet.getLastRow() + 1;
    sheet.getRange(row, 1, 1, 5).setValues([[
      salarie,
      "0h00",
      "0h00",
      "0h00",
      "0h00"
    ]]);
  }

  const values = sheet.getRange(row, 1, 1, 5).getValues()[0];
  return {
    aDevoir: Math.abs(convertirHeureEnMinutes(values[3] || "0h00")),
    heuresSup: Math.abs(convertirHeureEnMinutes(values[4] || "0h00"))
  };
}

function appliquerEcart(compteurs, ecart) {
  ecart = Number(ecart) || 0;

  if (ecart < 0) {
    let dette = Math.abs(ecart);

    if (compteurs.heuresSup > 0) {
      const compensation = Math.min(dette, compteurs.heuresSup);
      compteurs.heuresSup -= compensation;
      dette -= compensation;
    }

    compteurs.aDevoir += dette;
    return;
  }

  if (ecart <= 0) return;

  if (compteurs.aDevoir > 0) {
    const deduction = Math.min(ecart, compteurs.aDevoir);
    compteurs.aDevoir -= deduction;
    ecart -= deduction;
  }

  compteurs.heuresSup += ecart;
}

function getFeuilleId(data) {
  const salarie = normaliserTexte(data.salarie);
  const semaine = normaliserTexte(data.semaine);
  const dates = (data.lignes || [])
    .map(function(ligne) { return normaliserTexte(ligne.date); })
    .filter(Boolean)
    .sort()
    .join("_");

  return normaliserTexte(data.feuilleId) || [salarie, semaine, dates].join("|");
}

function validerSemaine(data) {
  // Validation serveur obligatoire: le navigateur ne doit jamais \u00eatre la seule s\u00e9curit\u00e9.
  if (!data || typeof data !== "object") {
    throw new Error("Donn\u00e9es de feuille invalides.");
  }

  const salarie = getNomSalarie(data.salarie);
  if (!salarie) {
    throw new Error("S\u00e9lectionne un salari\u00e9 valide avant l'envoi.");
  }

  if (!Array.isArray(data.lignes) || data.lignes.length === 0) {
    throw new Error("Aucune journ\u00e9e remplie \u00e0 envoyer.");
  }

  data.lignes.forEach(function(ligne) {
    const jour = normaliserTexte(ligne.jour) || "une journ\u00e9e";
    const statut = normaliserTexte(ligne.statut) || APP_CONFIG.presentStatus;

    if (APP_CONFIG.statuses.indexOf(statut) === -1) {
      throw new Error("Statut invalide pour " + jour + ".");
    }

    if (!normaliserTexte(ligne.date)) {
      throw new Error("Renseigne la date pour " + jour + ".");
    }

    if (statut !== APP_CONFIG.presentStatus) {
      return;
    }

    if (!normaliserTexte(ligne.chantier)) {
      throw new Error("Renseigne le chantier pour " + jour + ".");
    }

    const debut = convertirHoraireEnMinutes(ligne.debut);
    const fin = convertirHoraireEnMinutes(ligne.fin);
    const pause = Number(ligne.pauseMinutes);

    if (debut === null || fin === null) {
      throw new Error("Renseigne des horaires valides pour " + jour + ".");
    }

    if (fin <= debut) {
      throw new Error("L'heure de fin doit \u00eatre sup\u00e9rieure \u00e0 l'heure de d\u00e9but pour " + jour + ".");
    }

    if (!Number.isFinite(pause) || pause < 0) {
      throw new Error("La pause est invalide pour " + jour + ".");
    }

    if (fin - debut - pause < 0) {
      throw new Error("La pause d\u00e9passe le temps travaill\u00e9 pour " + jour + ".");
    }
  });

  data.salarie = salarie;
  data.feuilleId = getFeuilleId(data);
  return data;
}

function supprimerAncienneFeuille(sheet, feuilleId) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return 0;

  const ids = sheet.getRange(2, 2, lastRow - 1, 1).getValues();
  let deleted = 0;

  // Suppression de bas en haut pour conserver les bons num\u00e9ros de lignes.
  for (let i = ids.length - 1; i >= 0; i--) {
    if (ids[i][0] === feuilleId) {
      sheet.deleteRow(i + 2);
      deleted++;
    }
  }

  return deleted;
}

function migrerAnciennesSaisiesSiBesoin(ss, salarie, targetSheet) {
  const props = PropertiesService.getScriptProperties();
  const key = "migration_legacy_saisies_v2_" + salarie;

  if (props.getProperty(key)) return;

  const legacySheet = ss.getSheetByName(APP_CONFIG.legacySheetName);
  if (!legacySheet || legacySheet.getLastRow() < 2) {
    props.setProperty(key, "fait");
    return;
  }

  const values = legacySheet.getDataRange().getValues();
  const rows = [];

  values.forEach(function(row, index) {
    if (index === 0 || row[0] !== salarie) return;

    const jour = row[2] || "";
    const isMonthlyTotal = jour === "TOTAL MENSUEL";

    rows.push([
      salarie,
      "legacy|" + salarie + "|" + index,
      "",
      row[1] || "",
      jour,
      isMonthlyTotal ? "" : APP_CONFIG.presentStatus,
      row[3] || "",
      row[4] || "",
      row[5] || "",
      row[6] || "",
      row[7] || "",
      row[8] || "",
      row[9] || "",
      row[10] || "",
      row[11] || "",
      row[12] || ""
    ]);
  });

  if (rows.length) {
    targetSheet.getRange(targetSheet.getLastRow() + 1, 1, rows.length, APP_CONFIG.headers.length)
      .setValues(rows);
    ajouterSeparateurSemaine(targetSheet);
  }

  props.setProperty(key, "fait");
}

function buildRows(data) {
  const compteursSemaine = { aDevoir: 0, heuresSup: 0 };

  const rows = data.lignes.map(function(ligne) {
    const statut = normaliserTexte(ligne.statut) || APP_CONFIG.presentStatus;
    const present = statut === APP_CONFIG.presentStatus;
    const ecart = present ? convertirHeureEnMinutes(ligne.ecart || "0h00") : 0;

    if (present) {
      appliquerEcart(compteursSemaine, ecart);
    }

    return [
      data.salarie,
      data.feuilleId,
      data.semaine || "",
      ligne.date || "",
      ligne.jour || "",
      statut,
      present ? ligne.chantier || "" : "",
      present ? ligne.debut + " \u2192 " + ligne.fin : "",
      present ? ligne.pause || "" : "",
      present ? ligne.totalJour || "0h00" : "",
      present ? ligne.ecart || "0h00" : "",
      present ? formatHeures(compteursSemaine.aDevoir) : "",
      present ? formatHeures(compteursSemaine.heuresSup) : "",
      present ? ligne.distanceKm || "" : "",
      present ? ligne.zone || "" : "",
      ligne.note || ""
    ];
  });

  if (data.resume) {
    rows.push([
      data.salarie,
      data.feuilleId,
      data.semaine || "",
      "",
      "TOTAL SEMAINE",
      "",
      "",
      "",
      "",
      data.resume.totalTravaille || "0h00",
      "",
      data.resume.aDevoir || "0h00",
      data.resume.heuresSupplementaires || "0h00",
      "",
      "",
      "R\u00e9sum\u00e9 automatique de la semaine"
    ]);
  }

  return rows;
}

function recalculerSoldesDepuisOnglet(sheet, salarie) {
  // Le solde global est recalcul\u00e9 depuis les journ\u00e9es pr\u00e9sentes uniquement.
  // Les colonnes de l'onglet salari\u00e9 restent des compteurs de semaine, pas un cumul historique.
  const lastRow = sheet.getLastRow();
  const compteurs = { aDevoir: 0, heuresSup: 0 };

  if (lastRow < 2) {
    majSoldes(salarie, 0, 0);
    return compteurs;
  }

  const values = sheet.getRange(2, 1, lastRow - 1, APP_CONFIG.headers.length).getValues();

  values.forEach(function(row, index) {
    const rowSalarie = row[0];
    const jour = row[4];
    const statut = row[5];
    const ecart = row[10];

    if (rowSalarie === salarie && jour === "TOTAL MENSUEL") {
      compteurs.heuresSup = 0;
      return;
    }

    if (rowSalarie === salarie && jour === "TOTAL SEMAINE") {
      return;
    }

    if (rowSalarie !== salarie || statut !== APP_CONFIG.presentStatus) return;

    appliquerEcart(compteurs, convertirHeureEnMinutes(ecart));
  });

  majSoldes(salarie, compteurs.aDevoir, compteurs.heuresSup);
  return compteurs;
}

function parseSheetDate(value) {
  if (value instanceof Date) {
    return value;
  }

  const texte = normaliserTexte(value);
  if (!texte) return null;

  const iso = texte.match(/^(\d{4})-(\d{2})-(\d{2})$/);
  if (iso) {
    return new Date(Number(iso[1]), Number(iso[2]) - 1, Number(iso[3]));
  }

  const fr = texte.match(/^(\d{2})\/(\d{2})\/(\d{4})$/);
  if (fr) {
    return new Date(Number(fr[3]), Number(fr[2]) - 1, Number(fr[1]));
  }

  return null;
}

function calculerTotauxMensuelsSalarie(ss, salarie, mois, annee) {
  const sheet = getEmployeeSheet(ss, salarie);
  const lastRow = sheet.getLastRow();
  const totaux = { totalTravaille: 0, aDevoir: 0, heuresSup: 0 };

  if (lastRow < 2) return totaux;

  const values = sheet.getRange(2, 1, lastRow - 1, APP_CONFIG.headers.length).getValues();

  values
    .map(function(row, index) {
      return {
        index: index,
        row: row,
        date: parseSheetDate(row[3])
      };
    })
    .filter(function(item) {
      const row = item.row;
      const jour = row[4];
      const statut = row[5];

      if (row[0] !== salarie || statut !== APP_CONFIG.presentStatus || !item.date) return false;
      if (jour === "TOTAL SEMAINE" || jour === "TOTAL MENSUEL") return false;
      return item.date.getFullYear() === annee && item.date.getMonth() === mois;
    })
    .sort(function(a, b) {
      const diff = a.date.getTime() - b.date.getTime();
      return diff || a.index - b.index;
    })
    .forEach(function(item) {
      const row = item.row;
      const totalJour = convertirHeureEnMinutes(row[9]);
      const ecart = convertirHeureEnMinutes(row[10]);

      totaux.totalTravaille += Math.max(0, totalJour);
      appliquerEcart(totaux, ecart);
    });

  return totaux;
}

function supprimerTotauxMensuelsExistants(sheet, moisLabel) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return;

  const moisValues = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
  for (let i = moisValues.length - 1; i >= 0; i--) {
    if (moisValues[i][0] === moisLabel) {
      sheet.deleteRow(i + 2);
    }
  }
}

function enregistrerTotauxMensuels(ss, moisLabel, dateCalcul, resultats) {
  const sheet = getMonthlyTotalsSheet(ss);
  supprimerTotauxMensuelsExistants(sheet, moisLabel);

  const rows = resultats.map(function(resultat) {
    return [
      moisLabel,
      resultat.salarie,
      formatHeures(resultat.totalTravaille),
      formatHeures(resultat.aDevoir),
      formatHeures(resultat.heuresSup),
      dateCalcul
    ];
  });

  if (rows.length) {
    sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, 6).setValues(rows);
  }
}

function reparerOngletSalarie(ss, salarie) {
  const sheet = getEmployeeSheet(ss, salarie);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) {
    majSoldes(salarie, 0, 0);
    return;
  }

  sheet.getRange(1, 1, 1, APP_CONFIG.headers.length).setValues([APP_CONFIG.headers]);

  const range = sheet.getRange(2, 1, lastRow - 1, APP_CONFIG.headers.length);
  const values = range.getValues();
  let feuilleIdCourante = "";
  let compteursSemaine = { aDevoir: 0, heuresSup: 0 };
  let totalTravailleSemaine = 0;

  values.forEach(function(row) {
    const rowSalarie = row[0];
    const feuilleId = row[1];
    const jour = row[4];
    const statut = row[5];

    if (rowSalarie !== salarie || !feuilleId || rowSalarie === "--------------------") {
      row[11] = "";
      row[12] = "";
      return;
    }

    if (feuilleId !== feuilleIdCourante) {
      feuilleIdCourante = feuilleId;
      compteursSemaine = { aDevoir: 0, heuresSup: 0 };
      totalTravailleSemaine = 0;
    }

    if (jour === "TOTAL MENSUEL") {
      return;
    }

    if (jour === "TOTAL SEMAINE") {
      row[9] = formatHeures(totalTravailleSemaine);
      row[11] = compteursSemaine.aDevoir === 0 ? "0h00" : "\u2796 " + formatHeures(compteursSemaine.aDevoir);
      row[12] = formatHeures(compteursSemaine.heuresSup);
      return;
    }

    if (statut !== APP_CONFIG.presentStatus) {
      row[11] = "";
      row[12] = "";
      return;
    }

    totalTravailleSemaine += convertirHeureEnMinutes(row[9]);
    appliquerEcart(compteursSemaine, convertirHeureEnMinutes(row[10]));
    row[11] = formatHeures(compteursSemaine.aDevoir);
    row[12] = formatHeures(compteursSemaine.heuresSup);
  });

  range.setValues(values);
  recalculerSoldesDepuisOnglet(sheet, salarie);
}

function reparerCalculsHeures() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  ["J\u00e9r\u00e9my", "Damien"].forEach(function(salarie) {
    reparerOngletSalarie(ss, salarie);
  });
  recalculerTotauxMensuelsCourants();
}

function getLignesChronologiquesSalarie(ss, salarie) {
  const sheet = getEmployeeSheet(ss, salarie);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];

  const values = sheet.getRange(2, 1, lastRow - 1, APP_CONFIG.headers.length).getValues();

  return values
    .map(function(row, index) {
      return {
        index: index,
        salarie: row[0],
        date: parseSheetDate(row[3]),
        jour: row[4],
        statut: row[5],
        ecart: row[10]
      };
    })
    .filter(function(ligne) {
      if (ligne.salarie !== salarie) return false;
      if (!ligne.date) return false;
      if (ligne.jour === "TOTAL MENSUEL" || ligne.jour === "TOTAL SEMAINE") return false;
      if (ligne.salarie === "--------------------") return false;
      return ligne.statut === APP_CONFIG.presentStatus;
    })
    .sort(function(a, b) {
      const diff = a.date.getTime() - b.date.getTime();
      return diff || a.index - b.index;
    });
}

function recalculerSoldeSalarieDepuisDepart(ss, salarie) {
  const compteurs = getCompteursDepartSoldes(ss, salarie);
  const lignes = getLignesChronologiquesSalarie(ss, salarie);

  lignes.forEach(function(ligne) {
    appliquerEcart(compteurs, convertirHeureEnMinutes(ligne.ecart || "0h00"));
  });

  majSoldes(salarie, compteurs.aDevoir, compteurs.heuresSup);
  return {
    salarie: salarie,
    aDevoir: compteurs.aDevoir,
    heuresSup: compteurs.heuresSup,
    lignes: lignes.length
  };
}

function recalculerTousLesSoldes() {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const resultats = ["J\u00e9r\u00e9my", "Damien"].map(function(salarie) {
      reparerOngletSalarie(ss, salarie);
      return recalculerSoldeSalarieDepuisDepart(ss, salarie);
    });

    recalculerTotauxMensuelsCourants();

    return resultats.map(function(resultat) {
      return resultat.salarie + " : \u00c0 devoir " + formatHeures(resultat.aDevoir) +
        " | Heures sup " + formatHeures(resultat.heuresSup) +
        " (" + resultat.lignes + " lignes relues)";
    }).join("\n");
  } finally {
    lock.releaseLock();
  }
}

function enregistrerSemaine(data) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    data = validerSemaine(data);
    clotureMensuelleHeuresSup();

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = getEmployeeSheet(ss, data.salarie);

    migrerAnciennesSaisiesSiBesoin(ss, data.salarie, sheet);

    const feuillesRemplacees = supprimerAncienneFeuille(sheet, data.feuilleId);
    const rows = buildRows(data);

    if (rows.length) {
      sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, APP_CONFIG.headers.length)
        .setValues(rows);
      ajouterSeparateurSemaine(sheet);
    }

    const compteurs = recalculerSoldesDepuisOnglet(sheet, data.salarie);
    const prefix = feuillesRemplacees > 0 ? "Feuille remplac\u00e9e. " : "Feuille enregistr\u00e9e. ";

    return prefix + "\u00c0 devoir : " + formatHeures(compteurs.aDevoir) +
      " | Heures sup : " + formatHeures(compteurs.heuresSup);
  } finally {
    lock.releaseLock();
  }
}

function clotureMensuelleHeuresSup() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const today = new Date();
  const tomorrow = new Date(today);
  tomorrow.setDate(today.getDate() + 1);

  const isLastDayOfMonth = tomorrow.getMonth() !== today.getMonth();
  if (!isLastDayOfMonth) return;

  const timezone = Session.getScriptTimeZone();
  const mois = Utilities.formatDate(today, timezone, "MM/yyyy");
  const dateLigne = Utilities.formatDate(today, timezone, "yyyy-MM-dd");

  const props = PropertiesService.getScriptProperties();
  const cle = "cloture_heures_sup_" + mois;

  if (props.getProperty(cle)) return;

  const resultats = Object.keys(APP_CONFIG.employeeSheets)
    .filter(function(salarie, index, salaries) {
      return salarie !== "Jeremy" && salaries.indexOf(salarie) === index;
    })
    .map(function(salarie) {
      const totaux = calculerTotauxMensuelsSalarie(ss, salarie, today.getMonth(), today.getFullYear());
      return {
        salarie: salarie,
        totalTravaille: totaux.totalTravaille,
        aDevoir: totaux.aDevoir,
        heuresSup: totaux.heuresSup
      };
    });

  enregistrerTotauxMensuels(ss, mois, dateLigne, resultats);

  props.setProperty(cle, "fait");
}

function recalculerTotauxMensuelsCourants() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const today = new Date();
  const timezone = Session.getScriptTimeZone();
  const mois = Utilities.formatDate(today, timezone, "MM/yyyy");
  const dateLigne = Utilities.formatDate(today, timezone, "yyyy-MM-dd");

  const resultats = ["J\u00e9r\u00e9my", "Damien"].map(function(salarie) {
    const totaux = calculerTotauxMensuelsSalarie(ss, salarie, today.getMonth(), today.getFullYear());
    return {
      salarie: salarie,
      totalTravaille: totaux.totalTravaille,
      aDevoir: totaux.aDevoir,
      heuresSup: totaux.heuresSup
    };
  });

  enregistrerTotauxMensuels(ss, mois, dateLigne, resultats);
}

function installerDeclencheurClotureMensuelle() {
  ScriptApp.newTrigger("clotureMensuelleHeuresSup")
    .timeBased()
    .everyDays(1)
    .atHour(23)
    .create();
}
